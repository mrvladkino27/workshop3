CREATE OR REPLACE VIEW task as
SELECT 
       MANUFACTURER.COMPANY_NAME
       ,CAR.CAR_MODEL
       ,CAR.CAR_VIN
       ,CAR.CAR_YEAR
       ,FUEL.FUEL_NAME
      FROM
        CAR JOIN LOT ON CAR.CAR_VIN = lot.lot_vin
        JOIN CAR_MODEL ON CAR.CAR_MODEL = car_model.model_name
        JOIN MANUFACTURER ON CAR_MODEL.model_manufactured = MANUFACTURER.COMPANY_NAME
        JOIN FUEL ON CAR.CAR_FUEL = FUEL.FUEL_NAME
